## 单链表
链表由一些列的数据结构组成，每个结构包含表元素和该元素的后继元素的结构的指针(next)。最后一个元素的指针为null。

 在创建链表之前，我们需要需要了解一下专业术语：

-    首节点：存放第一个有效数据的节点；
-    尾节点：存放最后一个有效数据的节点；
-    头节点：头节点的数据类型与首节点的数据类型相同，并且头节点是首节点前面的那个节点，并不存放有效数据；头节点的存在只是为了方便链表的操作。
-    头指针：指向头节点的指针；
-    尾指针：指向尾节点的指针；
   
![image](http://images2015.cnblogs.com/blog/871366/201605/871366-20160505145921935-124469013.png)

```
 struct Node {
    
    int data;//存放数据
    
    struct Node *next;//指向下一个节点的结构体指针
}

typedef struct Node NODE; //定义一个Node类型
typedef struct Node  *Pnode; //定义一个结构体指针的类型

//创建一个头节点指针 指向头节点

Pnode head = (Pnode )malloc(sizeof(Node));

*head //头节点
head->next = NULL;//next节点指向NULL。

//创建第二个节点

Pnode Second = (Pnode) malloc (sizeof(Node));//

head->next = Second;//将head的next 指向Second
Second->data = 2;
Second->next = null

```

### 双向链表


![image](http://img.my.csdn.net/uploads/201212/11/1355190600_9761.jpg)

双向链表的特点。每个节点包含三个元素。指向前驱节点的指针、指向后续节点的指针和节点的值。

如果该节点没有前驱，则该节点是头节点。

如果该节点没有后续，则该节点是尾节点。

```
/*定义一个节点*/

typedef struct Node
{

    int data; // 节点的值

    struct Node *pre;//指向前驱指针

    struct Node *next; // 指向后续指针

} Node,*Pnode;

/*定义一个链表，包括头部和尾部节点和链表的长度*/
typedef struct list
{
    int length;
    struct Node *head;
    struct Node *tail;
} List,*Plist;

//初始化空链表

List *init()
{
    List *list =  (Plist) malloc(sizeof(Node));
    if(list == NULL)
        return NULL;
    list->head = NULL;
    list->tail = NULL;
    list->length = 0;
    return list;
}

//插入到链表的头部
void insertHead(List *list,int data)
{
    Node *node = malloc(sizeof(Node));
    if(node == NULL) {
        printf("error:内存不足");
        exit(-1);
    }

    node->data = data;
    // 判断是否是空链表。如果是空链表。把该元素设置头节点和尾节点
    if(list->length == 0)
    {
        list->head = list->tail = node;
        node->next = node->pre = NULL;

    }
    else
    {

        node->pre = NULL;
        node->next = list->head;
        list->head->pre = node;
        list->head = node; //修改列表头节点
    }
    
    list->length += 1;

}
// 插入到尾部节点

void insertTail(List *list,int data)
{
    Node *node = (Pnode) malloc(sizeof(Node));
    if(node == NULL) {
        printf("error:内存不足");
        exit(-1);
    }
    node->data = data;
    if(list->length == 0)
    {
        list->head = list->tail = node;
        node->pre = node->next = NULL;

    }
    else
    {


        node->next = NULL;
        node->pre = list->tail;
        list->tail->next = node; //修改尾节点
        list->tail = node;
    }
    list->length += 1;
}
//删除节点。
void deleteNode(List *list,Node *node)
{
    if(node->pre == NULL)
    {
        list->head = node->next;
        node->next->pre = NULL;
    }
    else
    {
        node->pre->next = node->next;

    }

    if(node->next == NULL)
    {
        list->tail = node->pre;
        node->pre->next = NULL;
    }
    else
    {
        node->next->pre = node->pre;
    }

    free(node);
    list->length--;

}
// 删除含有data数据的所有的节点。
void deleteNodes(List *list, int data)
{
    Node *head = list->head;

    while(head != list->tail)
    {
        if(head->data == data)
        {
            deleteNode(list,head);
        }
        head = head->next;
    }
    if(list->tail->data == data)
    {
        deleteNode(list,list->tail);
    }


}
//查找节点
Node *findNode(List *list,int data)
{
    Node *head = list->head;
    Node *node = head;
    if(list->tail->data == data)
        return list->tail;
    while(node != list->tail)
    {
        if(node->data == data)
            return node;
        node = node->next;
    }

    return NULL;

}

// 插入数据到对应的节点的前面或后面， direction 0表示前面，1 表示后面。

void insertData(List *list,Node *node,int data,int direction)
{
    Pnode new_node = (Pnode) malloc(sizeof(Node));
    if(node == NULL) {
        printf("error:内存不足");
        exit(-1);
    }
    new_node->data = data;
    // 插入前面
    if (direction == 0)
    {
        if(list->head == node){
            insertHead(list,data);
        }else{
            node->pre->next = new_node;
            new_node->pre = node->pre;
            new_node->next = node;
            node->pre = new_node;
        }
    }else{
        if(list->tail == node) {
            insertTail(list,data);
        }else{

            new_node->pre = node;
            new_node->next = node->next;
            node->next->pre = new_node;
            node->next = new_node;

        }

    }

}
// 打印所有节点。direction 表示方向
void printList(List *list,int direction)
{
    Node *node;
    Node *head = list->head;
    Node *tail = list->tail;
    if(direction == 1)
    {
        node = head;
        while(node != tail)
        {
            printf("%d<=>",node->data);
            node = node->next;
        }
        printf("%d\n",node->data);

    }
    else
    {

        node = tail;
        while(node != head)
        {
            printf("%d<=>",node->data);
            node = node->pre;
        }
        printf("%d\n",node->data);

    }
}
// 主函数

int main()
{
    List *list = init();
    insertHead(list,1);
    insertHead(list,2);
    insertTail(list,4);
    int t = 3;
    for (; t<10; t++)
    {
        insertTail(list,3);

    }
    Node *fnode = findNode(list,4);
    insertData(list,fnode,40,0);
    insertData(list,fnode,30,1);
    printList(list,1);
    deleteNodes(list,3);
    printList(list,1);
    return 0;
}
```
